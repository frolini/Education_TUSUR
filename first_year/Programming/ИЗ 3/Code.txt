/*Даны целые числа K, N, а также K наборов целых чисел по N элементов в каждом
наборе.Для каждого набора выполнить следующее действие : если в наборе со -
держится число 2, то вывести сумму его элементов; если в наборе нет двоек, то
вывести 0. Наборы генерировать.Массивы не использовать.*/

#include <stdio.h>
#include <stdlib.h>

void Sum(int K, int N)
{
	//Перебираем элементы
	srand(time(NULL));
	for (int i = 0; i < K; i++)
	{
		//Переменные суммы и флага
		int s = 0;
		int f = 0;
		for (int j = 0; j < N; j++)
		{
			//Генерируемм число
			int num = rand() % 100;
			//Ищем сумму
			s += num;
			//Если число = 2 - поднимаем флаг
			if (num == 2)
			{
				f = 1;
			}
			//Вывод для отладки
			printf("%4d ", num);
		}
		printf("\n");
		//Если флаг поднят - выводим сумму
		if (f)
		{
			printf("%d\n", s);
		}
		else 
		{
			printf("%d\n", 0);
		}
	
	}
}

int main()
{
	//Кол-во и размер наборов
	int K = 20;
	int N = 20;
	//Вызываем функцию
	Sum(K, N);
	getch();
	return 0;
}
